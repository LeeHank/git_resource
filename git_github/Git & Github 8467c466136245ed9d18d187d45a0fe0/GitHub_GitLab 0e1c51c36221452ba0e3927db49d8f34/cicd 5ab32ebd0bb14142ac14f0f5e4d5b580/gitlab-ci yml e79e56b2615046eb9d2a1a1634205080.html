stages:
  - deploy
  
deploy:
  stage: deploy
  
  script:
    - . /home/gitlab-runner/siteconfig
    - . /home/gitlab-runner/apconfig
    
    - echo $CI_PROJECT_NAME
    - eval SERVER_LOC=\${${CI_PROJECT_NAME}[0]}
    - eval REPOSRC=\${${CI_PROJECT_NAME}[1]}
    - GIT_SSH=${CI_PROJECT_PATH}.git
    
    - tmp=$CI_COMMIT_REF_NAME
    - tmpSITE=${tmp#release\/}
    - SITE=${tmpSITE%%\/*}
    - eval siteip="\$$SITE"
    - echo $SITE
    - echo $siteip
    - echo $GIT_SSH
    - echo $REPOSRC
    - echo --------------- start deploy ${SITE} ---------------
    - sshpass -p hadoop ssh ${siteip} 'bash -s' < /home/gitlab-runner/trigger.sh $SERVER_LOC $REPOSRC $GIT_SSH $CI_COMMIT_REF_NAME $SITE 
    - echo --------------- complete deploy ${SITE} ---------------
    
  only: 
    - /^release\/.*\/.*$/
    
  except:
    - /^release\/ALL\/.*$/

    
deploy_ALL:
  stage: deploy
  
  script:
    - . /home/gitlab-runner/siteconfig
    - . /home/gitlab-runner/apconfig
    
    - eval SERVER_LOC=\${${CI_PROJECT_NAME}[0]}
    - eval REPOSRC=\${${CI_PROJECT_NAME}[1]}
    - GIT_SSH=${CI_PROJECT_PATH}.git

    - for SITE in ${ALL[@]};
      do eval siteip="\$$SITE";
      echo --------------- start deploy ${SITE} ---------------;
      sshpass -p hadoop ssh ${siteip} 'bash -s' < /home/gitlab-runner/trigger.sh $SERVER_LOC $REPOSRC $GIT_SSH $CI_COMMIT_REF_NAME $SITE;
      echo --------------- complete deploy ${SITE} ---------------;
      done
    
  only: 
    - /^release\/ALL\/.*$/

  • 可以看到 only 這邊，他會看你的 tag，來決定 run 哪一段。這邊他下的 tag 是 release/ALL/1.3.0 ，所以是 run 下面那段
  • 這份文件的執行環境，是我們 註冊 gitlab-runner 的那台主機。而當初我們是註冊在172.16.56.190 的gitlab-runner 這個帳號下 (如果 gitlab 不是架在自己家的 server 也沒差，註冊gitlab-runner 時註冊自己所在的主機就好)。所以，一開始的 . /home/gitlab-runner/siteconfig ，是去讀各 site 的資訊，內容如下：

export TEST=hadoop@172.16.99.233
export WS1=hadoop@10.59.100.231
export WS2=hadoop@10.69.100.231
export KS=hadoop@172.25.200.160
export VN2=hadoop@10.169.96.14
  • 也就是把這些環境變數給 export 到 global environment

  • 然後就開始用迴圈，ssh到各site中執行指令。重要的是這段： sshpass -p hadoop ssh ${siteip} 'bash -s' < /home/gitlab-runner/trigger.sh $SERVER_LOC $REPOSRC $GIT_SSH $CI_COMMIT_REF_NAME $SITE; 意思是：
    • -p hadoop 密碼是 hadoop
    • ssh ${siteip} 我要 ssh 到這台 server ip，例如：hadoop@172.16.99.233 ，所以帳號也是 hadoop
    • bash -s 我要執行 bash 指令，執行啥指令？執行後面這段： /home/gitlab-runner/trigger.sh $SERVER_LOC $REPOSRC $GIT_SSH

  • trigger.sh
#!/bin/sh

SERVER_LOC=$1 #/home/hadoop/EDSA/Code_App/R_App
REPOSRC=$2    #ClearLog
GIT_SSH='git@172.16.56.190:'$3  #git@172.16.56.80:edsa/Other_App/ClearLog.git
head_tags=$4
# We do it this way so that we can abstract if from just git later on
site=$5

LOCALREPO_VC_DIR=$SERVER_LOC'/'$REPOSRC

echo Path : $LOCALREPO_VC_DIR
echo head_tags : $head_tags
echo GIT_SSH : $GIT_SSH

if [ ! -d $LOCALREPO_VC_DIR ]
then
	echo Create
	pwd 
	mkdir -p $SERVER_LOC
	cd
	cd $SERVER_LOC
	echo clone $GIT_SSH
    git clone --no-checkout $GIT_SSH $REPOSRC 
	cd $REPOSRC
	git fetch origin --prune
	git reset --hard 
	git checkout $head_tags
	
else
	echo Update
	cd
	cd $LOCALREPO_VC_DIR
	#git tag -l | xargs git tag -d
	git fetch origin --prune
	git reset --hard 
	git checkout $head_tags
fi

if [ "${site}" != "TEST" ];then chmod 775 $LOCALREPO_VC_DIR/*;else chmod 777 $LOCALREPO_VC_DIR/*;fi
# 改為所有人可執行，確保 crontab 可 run

# 確認有沒有這個，有的話表示是 renv 專案，需修改 R.version$platform 並 restore才能使用 package
FILE=renv/activate.R
if test -f "$FILE"; then
	sed -i '3i R.version$platform<- "x86_64-redhat-linux-gnu"' 'renv/activate.R'
	sed -i '25i library(renv, lib.loc = .Library)' 'renv/activate.R'
	R -e 'renv::restore()'
fi
